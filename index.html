ขั้นตอนที่ 2 — วาง Code นี้ใน Code.gs

ลบ code เดิมออกทั้งหมด แล้ววางอันนี้แทน:



const SPREADSHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();

const SCORES_SHEET_NAME = "คะแนนรวม";

 

function doGet(e) {

  return handleRequest(e);

}

function doPost(e) {

  return handleRequest(e);

}

 

function handleRequest(e) {

  const callback = e.parameter.callback;

  try {

    const action = e.parameter.action || (e.postData ? JSON.parse(e.postData.contents).action : null);

    

    if (action === "saveScore") {

      const data = e.postData ? JSON.parse(e.postData.contents) : e.parameter;

      const result = saveScore(data);

      return output(callback, result);

    }

    if (action === "getScores") {

      const result = getAllScores();

      return output(callback, result);

    }

    if (action === "getStudents") {

      const result = getStudents();

      return output(callback, result);

    }

    return output(callback, {success: false, error: "Unknown action"});

  } catch(err) {

    return output(callback, {success: false, error: err.toString()});

  }

}

 

function output(callback, data) {

  const json = JSON.stringify(data);

  if (callback) {

    return ContentService.createTextOutput(callback + "(" + json + ")")

      .setMimeType(ContentService.MimeType.JAVASCRIPT);

  }

  return ContentService.createTextOutput(json)

    .setMimeType(ContentService.MimeType.JSON);

}

 

function saveScore(data) {

  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

  let sheet = ss.getSheetByName(SCORES_SHEET_NAME);

  

  // สร้าง sheet ใหม่ถ้ายังไม่มี

  if (!sheet) {

    sheet = ss.insertSheet(SCORES_SHEET_NAME);

    const headers = [

      "timestamp", "evaluator", "studentId", "studentName", "topic",

      "s1q1", "s1q2", "s1q3", "s1_total",

      "s2q1", "s2q2", "s2q3", "s2q4", "s2q5", "s2_total",

      "s3q1", "s3_total",

      "grand_total"

    ];

    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

    sheet.getRange(1, 1, 1, headers.length).setFontWeight("bold");

  }

  

  // ค้นหาแถวที่มีอาจารย์และนิสิตนี้อยู่แล้ว

  const allData = sheet.getDataRange().getValues();

  let targetRow = -1;

  for (let i = 1; i < allData.length; i++) {

    if (allData[i][1] === data.evaluator && allData[i][2] === data.studentId) {

      targetRow = i + 1;

      break;

    }

  }

  

  const row = [

    new Date().toISOString(),

    data.evaluator,

    data.studentId,

    data.studentName,

    data.topic || "",

    data.s1q1 || 0, data.s1q2 || 0, data.s1q3 || 0, data.s1_total || 0,

    data.s2q1 || 0, data.s2q2 || 0, data.s2q3 || 0, data.s2q4 || 0, data.s2q5 || 0, data.s2_total || 0,

    data.s3q1 || 0, data.s3_total || 0,

    data.grand_total || 0

  ];

  

  if (targetRow > 0) {

    sheet.getRange(targetRow, 1, 1, row.length).setValues([row]);

  } else {

    sheet.appendRow(row);

  }

  

  return {success: true, message: "บันทึกสำเร็จ", row: targetRow > 0 ? targetRow : sheet.getLastRow()};

}

 

function getAllScores() {

  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

  const sheet = ss.getSheetByName(SCORES_SHEET_NAME);

  if (!sheet) return {success: true, scores: []};

  const data = sheet.getDataRange().getValues();

  if (data.length <= 1) return {success: true, scores: []};

  const headers = data[0];

  const scores = data.slice(1).map(row => {

    const obj = {};

    headers.forEach((h, i) => obj[h] = row[i]);

    return obj;

  });

  return {success: true, scores};

}

 

function getStudents() {

  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

  const sheet = ss.getSheetByName("รายชื่อนิสิต");

  if (!sheet) return {success: true, students: []};

  const data = sheet.getDataRange().getValues();

  const students = data.slice(1).filter(r => r[1]).map(r => ({

    order: r[0], id: String(r[1]), name: r[2], topic: r[3], 

    abstract: r[4], date: r[9]

  }));

  return {success: true, students};

}
 

ขั้นตอนที่ 3 — Deploy เป็น Web App

1. คลิก Deploy → New deployment
2. เลือก Type: Web app
3. Execute as: Me
4. Who has access: Anyone (หรือ "Anyone within [organization]")
5. คลิก Deploy → Copy URL ที่ได้

 

ขั้นตอนที่ 4 — ใส่ URL ในระบบนี้

กลับมาที่หน้านี้ → คลิก ⚙ ตั้งค่า มุมขวาบน → วาง URL ที่ได้จาก Apps Script

 
⚙ ตั้งค่าการเชื่อมต่อ
ใส่ URL ของ Google Apps Script Web App เพื่อเชื่อมต่อกับ Google Sheets โดยตรง
ดูวิธีสร้าง Web App ได้ที่แท็บ "วิธีตั้งค่า Apps Script"
