// ─── LOAD STUDENTS FROM SHEET ────────────────────────────────────────


function loadStudentsFromSheet() {


  if (!state.scriptUrl) return;


  const url = state.scriptUrl + '?action=getStudents&callback=handleStudents';


  const script = document.createElement('script');


  script.src = url;


  document.body.appendChild(script);


  setTimeout(() => script.remove(), 5000);


}

    renderStudentList();


    updateStatus('connected');


  }


};

    return s.id.toLowerCase().includes(query) || s.name.toLowerCase().includes(query);


  });


 


  // Update search count badge


  const countEl = document.getElementById('searchCount');


  if (countEl) {


    if (query) {


      countEl.style.display = 'block';


      countEl.textContent = `พบ ${filtered.length} / ${state.students.length} คน`;


    } else {


      countEl.style.display = 'none';


    }


  }


 


  if (filtered.length === 0) {


    list.innerHTML = `<div style="padding:16px; text-align:center; color:var(--text-dimmer); font-size:13px;">ไม่พบนิสิตที่ค้นหา</div>`;


    return;


  }


 


  filtered.forEach(s => {


    const key = getScoreKey(state.evaluator, s.id);


    const sc = state.scores[key] || {};


    const s1 = sc.s1_total || 0, s2 = sc.s2_total || 0, s3 = sc.s3_total || 0;


    const isActive = state.selectedStudent && state.selectedStudent.id === s.id;


 


    // Highlight matching text


    const nameDisplay = query ? highlightText(s.name, query) : s.name;


    const idDisplay = query ? highlightText(s.id, query) : s.id;


 


    const div = document.createElement('div');


    div.className = 'student-item' + (isActive ? ' active' : '');


    div.onclick = () => selectStudent(s);


    div.innerHTML = `


      <div class="student-name">${nameDisplay}</div>


      <div class="student-id">${idDisplay}</div>


      <div class="student-score-badge">


        <span class="score-pill ${s1 > 0 ? 'p1' : 'empty'}">ส1: ${s1}</span>


        <span class="score-pill ${s2 > 0 ? 'p2' : 'empty'}">ส2: ${s2}</span>


        <span class="score-pill ${s3 > 0 ? 'p3' : 'empty'}">ส3: ${s3}</span>


      </div>


    `;


    list.appendChild(div);


  });


}


 


function highlightText(text, query) {


  if (!query) return text;


  const regex = new RegExp(`(${query.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')})`, 'gi');


  return text.replace(regex, '<mark style="background:rgba(96,165,250,0.3); color:var(--accent2); border-radius:2px; padding:0 1px;">$1</mark>');


}

    </div>


  `;

function levelMatchFor(val, criterion, levelIndex) {


  const lv = criterion.levels[levelIndex];


  if (levelIndex === criterion.levels.length - 1) return val <= lv.max && val >= 0;


  if (levelIndex === 0) return val >= lv.min && val <= lv.max;


  return val >= lv.min && val <= lv.max;


}


 


function selectLevel(criterionId, levelIdx, cardEl, sectionId) {


  const key = getScoreKey(state.evaluator, state.selectedStudent.id);


  const sc = getOrInitScores(key);


  const sec = SECTIONS[sectionId];


  const criterion = sec.criteria.find(c => c.id === criterionId);


  const lv = criterion.levels[levelIdx];


  


  // Set default to midpoint of range


  let defaultVal;


  if (lv.min === 0 && lv.max === 0) defaultVal = 0;


  else defaultVal = Math.round((lv.min + lv.max) / 2 * 10) / 10;


  


  sc[criterionId] = defaultVal;


  const inputEl = document.getElementById('input_' + criterionId);


  if (inputEl) { inputEl.value = defaultVal; inputEl.className = 'score-number-input valid'; }


  recalcSection(sc, sectionId);


  updateSectionTotal(sc, sectionId);


  updateSummaryPanel(sc);


  


  // Update card highlights


  const allCards = cardEl.parentElement.querySelectorAll('.level-card');


  allCards.forEach((c, i) => {


    c.className = 'level-card';


    if (i === levelIdx) c.classList.add(`selected-${criterion.levels[i].tag}`);


  });


}


 


function onScoreInput(criterionId, val, sectionId) {


  const key = getScoreKey(state.evaluator, state.selectedStudent.id);


  const sc = getOrInitScores(key);


  const numVal = parseFloat(val);


  const sec = SECTIONS[sectionId];


  const criterion = sec.criteria.find(c => c.id === criterionId);


  const maxVal = criterion.levels[0].max;


  


  const inputEl = document.getElementById('input_' + criterionId);


  if (isNaN(numVal) || numVal < 0 || numVal > maxVal) {


    inputEl.className = 'score-number-input invalid';


    return;


  }


  inputEl.className = 'score-number-input valid';


  sc[criterionId] = numVal;


  recalcSection(sc, sectionId);


  updateSectionTotal(sc, sectionId);


  updateSummaryPanel(sc);


  


  // Highlight matching level card


  const container = inputEl.closest('.rubric-row');


  if (container) {


    const cards = container.querySelectorAll('.level-card');


    cards.forEach((card, i) => {


      card.className = 'level-card';


      const lv = criterion.levels[i];


      const match = (i === criterion.levels.length - 1) ? (numVal >= 0 && numVal <= lv.max) : (numVal >= lv.min && numVal <= lv.max);


      if (match) card.classList.add(`selected-${lv.tag}`);


    });


  }


}


 


function recalcSection(sc, sectionId) {


  const sec = SECTIONS[sectionId];


  let total = 0;


  sec.criteria.forEach(c => { if (sc[c.id] !== null) total += (sc[c.id] || 0); });


  sc[sectionId + '_total'] = Math.round(total * 10) / 10;


  sc.grand_total = Math.round(((sc.s1_total||0) + (sc.s2_total||0) + (sc.s3_total||0)) * 10) / 10;


}


 


function updateSectionTotal(sc, sectionId) {

}

// ─── SAVE TO SHEET ────────────────────────────────────────────────────


function saveToSheet() {


  if (!state.scriptUrl) { alert('กรุณาตั้งค่า Apps Script URL ก่อน'); return; }


  if (!state.evaluator || !state.selectedStudent) { alert('กรุณาเลือกผู้ประเมินและนิสิตก่อน'); return; }


  


  const key = getScoreKey(state.evaluator, state.selectedStudent.id);


  const sc = state.scores[key] || {};


  const saveBtn = document.getElementById('saveBtn');


  const fb = document.getElementById('saveFeedback');


  


  updateStatus('saving');


  saveBtn.disabled = true;


  saveBtn.textContent = '⏳ กำลังบันทึก...';


 


  const payload = {


    action: 'saveScore',


    evaluator: state.evaluator,


    studentId: state.selectedStudent.id,


    studentName: state.selectedStudent.name,


    topic: state.selectedStudent.topic || '',


    ...sc


  };


 


  // JSONP call


  const cbName = 'saveCallback_' + Date.now();


  window[cbName] = function(data) {


    saveBtn.disabled = false;


    saveBtn.textContent = '💾 บันทึกลง Google Sheets';


    fb.style.display = 'block';


    if (data.success) {


      updateStatus('connected');


      fb.className = 'save-feedback success';


      fb.textContent = '✅ บันทึกสำเร็จ — ' + new Date().toLocaleTimeString('th-TH');


    } else {


      updateStatus('disconnected');


      fb.className = 'save-feedback error';


      fb.textContent = '❌ เกิดข้อผิดพลาด: ' + (data.error || 'Unknown');


    }


    delete window[cbName];


    setTimeout(() => { if(fb) fb.style.display = 'none'; }, 4000);


  };


 


  const qs = Object.entries(payload).map(([k,v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v)}`).join('&');


  const script = document.createElement('script');


  script.src = state.scriptUrl + '?' + qs + '&callback=' + cbName;


  script.onerror = () => {


    saveBtn.disabled = false;


    saveBtn.textContent = '💾 บันทึกลง Google Sheets';


    fb.style.display = 'block';


    fb.className = 'save-feedback error';


    fb.textContent = '❌ ไม่สามารถเชื่อมต่อได้ กรุณาตรวจสอบ URL';


    updateStatus('disconnected');


    delete window[cbName];


  };


  document.body.appendChild(script);


  setTimeout(() => script.remove(), 8000);


}
